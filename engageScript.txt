create table ADMINISTRADOR
(

  LOGIN           VARCHAR(20) not null,
  SENHA           VARCHAR(20) not null,
  PRIMARY KEY (`IDADMINISTRADOR`) 
) ENGINE = InnoDB;

-- Create table
create table AREA
(
  IDAREA INT UNSIGNED NOT NULL AUTO_INCREMENT,
  AREA   VARCHAR(50),
  PRIMARY KEY (`IDAREA`) 
) ENGINE = InnoDB;

-- Create table
create table USUARIO
(
  IDUSUARIO   INT UNSIGNED NOT NULL AUTO_INCREMENT,
  AREA        INT UNSIGNED NOT NULL,
  NOMEUSUARIO VARCHAR(50),
  EMAIL       VARCHAR(50),
  PRIMARY KEY (`IDUSUARIO`) ,
  INDEX `IDX_USUARIO` (`AREA` ASC) ,
  CONSTRAINT `FK_USUARIO`
    FOREIGN KEY (`AREA` )
    REFERENCES `AREA` (`IDAREA` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;

create table TIPOPROGRAMA
(

  TIPOPROGRAMA   VARCHAR(20),
  PRIMARY KEY (`IDTIPOPROGRAMA`) 
) ENGINE = InnoDB;

--
-- Create table
create table PROGRAMA
(
  IDPROGRAMA   INT UNSIGNED NOT NULL AUTO_INCREMENT,
  NOMEPROGRAMA VARCHAR(50) NOT NULL,
  TIPOPROGRAMA INT UNSIGNED NOT NULL,
  PRIMARY KEY (`IDPROGRAMA`) ,
  INDEX `IDX_PROGRAMA_TIPOPROGRAMA` (`TIPOPROGRAMA` ASC) ,
  CONSTRAINT `FK_PROGRAMA_TIPOPROGRAMA`
    FOREIGN KEY (`TIPOPROGRAMA` )
    REFERENCES `TIPOPROGRAMA` (`IDTIPOPROGRAMA` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;

-- Create table
create table PONTUACAO
(
  IDPONTUACAO INT UNSIGNED NOT NULL AUTO_INCREMENT,
  USUARIO     INT UNSIGNED NOT NULL,
  PROGRAMA    INT UNSIGNED NOT NULL,
  DATA        DATE,
  PONTOS      INT UNSIGNED,
  BONUS       INT UNSIGNED,
  INDEX `IDX_PONTUACAOPROGRAMA` (`PROGRAMA` ASC) ,
  INDEX `IDX_PONTUACAOUSUARIO` (`USUARIO` ASC) ,
  PRIMARY KEY (`IDPONTUACAO`) ,
  CONSTRAINT `FK_PONTUACAOPROGRAMA`
    FOREIGN KEY (`PROGRAMA` )
    REFERENCES `PROGRAMA` (`IDPROGRAMA` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `FK_PONTUACAOUSUARIO`
    FOREIGN KEY (`USUARIO` )
    REFERENCES `USUARIO` (`IDUSUARIO` )
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;
